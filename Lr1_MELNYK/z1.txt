#include <iostream>

using namespace std;

int random(int min, int max)
{
   static bool first = true;
   if (first) 
   {  
      srand( time(NULL) ); 
      first = false;
   }
   return min + rand() % (( max + 1 ) - min);
}

int main()
{
    int n; 
    cin>>n;
    int *arr=new int[n];
    for(int i=0; i<n; ++i)
    {
        arr[i]=random(-100,100);
        cout<<arr[i]<<" ";
    }
    int sum=0;
    int dob=1;
    for(int i=0; i<n; ++i)
    {
        if(i%2==0)
        {
            sum+=arr[i];
        }
        else
        {
            dob*=arr[i];
        }
    }
    cout<<"\nSum = "<<sum<<" Dob = "<<dob;
    return 0;
}
